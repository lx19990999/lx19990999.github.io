uos1050e升级gcc

wget https://ftp.gnu.org/gnu/gcc/gcc-8.3.0/gcc-8.3.0.tar.xz
tar -xf gcc-8.3.0.tar.xz
cd gcc-8.3.0
./contrib/download_prerequisites
mkdir build && cd build
../configure –disable-multilib –enable-languages=c,c++
make -j$(nproc) && sudo make install > compile2.log 2>&1

sudo cp ./libstdc++.so.6.0.25 /usr/local/lib64/
sudo ln -sf /usr/local/lib64/libstdc++.so.6.0.25 /usr/local/lib64/libstdc++.so.6
sudo ldconfig

strings /usr/local/lib64/libstdc++.so.6 | grep GLIBCXX_3.4.25
